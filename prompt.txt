import { create } from 'zustand';
import { devtools } from 'zustand/middleware';
import { DeviceType, Client, DepositStatus } from '@/types';

interface RepairFormData {
  deviceType: DeviceType | null;
  brand: string | null;
  model: string | null;
  issues: string[];
  issueDescription: string;
  accessories: string[];
  password: string;
  depositStatus: DepositStatus;
  client: Client | null;
  newClient: {
    name: string;
    phone: string;
    email: string;
  };
  totalPrice: number;
  scheduledDate?: Date;
}

interface AddReparationStore {
  formData: RepairFormData;
  currentStep: number;
  setFormData: (data: Partial<RepairFormData>) => void;
  setStep: (step: number) => void;
  resetForm: () => void;
  nextStep: () => void;
  prevStep: () => void;
  submitForm: () => Promise<void>;
}

          name: '',
          phone: '',
          email: '',

        totalPrice: 0,

'use client';

import { useState, useEffect } from 'react';
import { Button } from '@/components/ui/button';
import { Input } from '@/components/ui/input';
import { Label } from '@/components/ui/label';
import { Checkbox } from '@/components/ui/checkbox';
import { Textarea } from '@/components/ui/textarea';
import { RadioGroup, RadioGroupItem } from '@/components/ui/radio-group';
import {
  Smartphone,
  Laptop,
  Tablet,
  Package,
  Check,
  AlertCircle,
  Search,
  CalendarIcon,
  Plus,
  X,
  ChevronRight,
} from 'lucide-react';
import { Calendar } from '@/components/ui/calendar';
import {
  Popover,
  PopoverContent,
  PopoverTrigger,
} from '@/components/ui/popover';
import { format } from 'date-fns';
import { fr } from 'date-fns/locale';
import {
  brands,
  models,
  issues,
  mockClients,
  mockParts,
  deviceTypes,
} from '@/lib/data';
import type { DeviceType, Client, DepositStatus } from '@/types';
import { cn } from '@/lib/utils';
import { Badge } from '@/components/ui/badge';
import { Separator } from '@/components/ui/separator';
import { useAddReparationStore } from '@/store/addReparationStore';
import { useRouter } from 'next/navigation';

interface IssuesStepProps {
  onFormSubmit?: () => void;
}

export function IssuesStep({ onFormSubmit }: IssuesStepProps) {
  const { formData, setFormData } = useAddReparationStore();
  const router = useRouter();

  const [selectedIssues, setSelectedIssues] = useState<string[]>(formData.issues);
  const [issueDescription, setIssueDescription] = useState(formData.issueDescription);
  const [accessories, setAccessories] = useState<string[]>(formData.accessories);
  const [newAccessory, setNewAccessory] = useState("");
  const [password, setPassword] = useState(formData.password);
  const [depositStatus, setDepositStatus] = useState<DepositStatus>(formData.depositStatus);
  const [scheduledDate, setScheduledDate] = useState<Date | undefined>(formData.scheduledDate);
  const [partsAvailability, setPartsAvailability] = useState<Record<string, boolean>>({});

  useEffect(() => {
    if (formData.issues) setSelectedIssues(formData.issues);
    if (formData.issueDescription) setIssueDescription(formData.issueDescription);
    if (formData.accessories) setAccessories(formData.accessories);
    if (formData.password) setPassword(formData.password);
    if (formData.depositStatus) setDepositStatus(formData.depositStatus);
    if (formData.scheduledDate) setScheduledDate(formData.scheduledDate);
  }, [formData.issues, formData.issueDescription, formData.accessories, formData.password, formData.depositStatus, formData.scheduledDate]);

  const filteredIssues = formData.deviceType
    ? issues.filter((i) => i.deviceTypes.includes(formData.deviceType as DeviceType))
    : [];

  const checkPartsAvailability = (issueName: string) => {
    const issue = issues.find((i) => i.name === issueName);
    if (!issue?.requiresPart) return true;

    const availablePart = mockParts.find(
      (part) =>
        part.name
          .toLowerCase()
          .includes(issueName.toLowerCase().split(" ")[0]) && part.quantity > 0,
    );
    return !!availablePart;
  };

  useEffect(() => {
    const availability: Record<string, boolean> = {};
    selectedIssues.forEach(issueName => {
      availability[issueName] = checkPartsAvailability(issueName);
    });
    setPartsAvailability(availability);
  }, [selectedIssues]);

  const handleIssueToggle = (issueName: string, checked: boolean) => {
    if (checked) {
      setSelectedIssues([...selectedIssues, issueName]);
    } else {
      setSelectedIssues(selectedIssues.filter((i) => i !== issueName));
    }
  };

  const handleAddAccessory = () => {
    if (newAccessory.trim()) {
      setAccessories([...accessories, newAccessory.trim()]);
      setNewAccessory("");
    }
  };

  const handleRemoveAccessory = (index: number) => {
    setAccessories(accessories.filter((_, i) => i !== index));
  };

  const calculateTotalPrice = () => {
    let total = 0;
    selectedIssues.forEach(issueName => {
      const issue = issues.find(i => i.name === issueName);
      if (issue) total += issue.basePrice;
    });
    return total;
  };

  const handleNext = () => {
    const totalPrice = calculateTotalPrice();
    setFormData({
      issues: selectedIssues,
      issueDescription,
      accessories,
      password,
      depositStatus,
      scheduledDate,
      totalPrice
    });
    // Navigate to the next step
    router.push('/add-reparation/client');
  };

  const handlePrev = () => {
    const totalPrice = calculateTotalPrice();
    setFormData({
      issues: selectedIssues,
      issueDescription,
      accessories,
      password,
      depositStatus,
      scheduledDate,
      totalPrice
    });
    // Navigate to the previous step
    router.push('/add-reparation/device');
  };

  return (
    <>
      <div className="flex items-center justify-between mb-6">
        <div>
          <h2 className="text-xl font-semibold text-card-foreground">
            Pannes & Réparations
          </h2>
          <p className="text-sm text-muted-foreground mt-1">
            Sélectionnez toutes les réparations nécessaires
          </p>
        </div>
        {selectedIssues.length > 0 && (
          <Badge variant="default" className="gap-1.5">
            <Check className="h-3.5 w-3.5" />
            {selectedIssues.length} sélectionnée
            {selectedIssues.length > 1 ? "s" : ""}
          </Badge>
        )}
      </div>

      <div className="space-y-2 mb-6">
        {filteredIssues.map((issue) => {
          const isSelected = selectedIssues.includes(issue.name);
          const isAvailable = partsAvailability[issue.name];
          return (
            <div
              key={issue.id}
              className={cn(
                "flex items-start gap-3 rounded-lg border p-4 transition-all",
                isSelected
                  ? "border-primary bg-primary/5"
                  : "border-border hover:border-primary/50 hover:bg-muted/50",
              )}
            >
              <Checkbox
                id={issue.id}
                checked={isSelected}
                onCheckedChange={(checked) =>
                  handleIssueToggle(issue.name, checked as boolean)
                }
                className="mt-0.5"
              />
              <label
                htmlFor={issue.id}
                className="flex-1 cursor-pointer"
              >
                <div className="flex items-center justify-between gap-2 mb-1">
                  <span className="font-medium text-card-foreground">
                    {issue.name}
                  </span>
                  <span className="text-sm font-semibold text-accent">
                    {issue.basePrice} €
                  </span>
                </div>
                <div className="flex items-center gap-2 flex-wrap">
                  {issue.requiresPart && (
                    <span className="text-xs text-muted-foreground">
                      Pièce requise
                    </span>
                  )}
                  {issue.requiresPart && isSelected && (
                    <Badge
                      variant={
                        isAvailable ? "default" : "destructive"
                      }
                      className="text-xs gap-1"
                    >
                      {isAvailable ? (
                        <>
                          <Check className="h-3 w-3" />
                          En stock
                        </>
                      ) : (
                        <>
                          <AlertCircle className="h-3 w-3" />À
                          commander
                        </>
                      )}
                    </Badge>
                  )}
                </div>
              </label>
            </div>
          );
        })}
      </div>

      <div className="space-y-2 mb-6">
        <Label
          htmlFor="issue-description"
          className="text-sm font-medium text-card-foreground"
        >
          Description détaillée (optionnel)
        </Label>
        <Textarea
          id="issue-description"
          placeholder="Ajoutez des détails supplémentaires sur les problèmes rencontrés..."
          value={issueDescription}
          onChange={(e) => setIssueDescription(e.target.value)}
          rows={3}
          className="resize-none"
        />
      </div>

      <div className="space-y-6">
        {/* Accessories */}
        <div className="space-y-3">
          <Label
